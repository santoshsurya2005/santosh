Stack Code


// Stack Using Array
#include <stdio.h>
#define MAX 100

int stack[MAX];
int top = -1;

void push(int value) {
    if (top == MAX - 1) {
        printf("Stack Overflow\\n");
    } else {
        stack[++top] = value;
        printf("%d pushed to stack\\n", value);
    }
}

int pop() {
    if (top == -1) {
        printf("Stack Underflow\\n");
        return -1;
    } else {
        return stack[top--];
    }
}

int peek() {
    if (top == -1) {
        printf("Stack is Empty\\n");
        return -1;
    } else {
        return stack[top];
    }
}

int isEmpty() {
    return top == -1;
}










// Stack Using Pointers
#include 

struct Node {
    int data;
    struct Node* next;
};

struct Node* topPtr = NULL;

void pushPtr(int value) {
    struct Node* newNode = (struct Node*)malloc(sizeof(struct Node));
    newNode->data = value;
    newNode->next = topPtr;
    topPtr = newNode;
    printf("%d pushed to stack\\n", value);
}

int popPtr() {
    if (!topPtr) {
        printf("Stack Underflow\\n");
        return -1;
    } else {
        int poppedValue = topPtr->data;
        struct Node* temp = topPtr;
        topPtr = topPtr->next;
        free(temp);
        return poppedValue;
    }
}

int peekPtr() {
    if (!topPtr) {
        printf("Stack is Empty\\n");
        return -1;
    } else {
        return topPtr->data;
    }
}

int isEmptyPtr() {
    return topPtr == NULL;
}

int main() {
    // Example usage for both implementations
    push(10);
    push(20);
    printf("Top element is %d\\n", peek());
    printf("%d popped from stack\\n", pop());

    pushPtr(30);
    pushPtr(40);
    printf("Top element using pointer stack is %d\\n", peekPtr());
    printf("%d popped from pointer stack\\n", popPtr());

    return 0;
}
