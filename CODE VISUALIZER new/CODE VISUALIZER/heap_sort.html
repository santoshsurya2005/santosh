#include <stdio.h>

void swap(int* a, int* b) {
  int t = *a; *a = *b; *b = t;
}

void heapify(int arr[], int n, int i) {
  int largest = i;
  int l = 2*i+1;
  int r = 2*i+2;

  if (larr[largest])
    largest = l;
  if (rarr[largest])
    largest = r;

  if (largest != i) {
    swap(&arr[i], &arr[largest]);
    heapify(arr, n, largest);
  }
}

void heapSort(int arr[], int n) {
  for (int i=n/2-1;i>=0;i--)
    heapify(arr,n,i);
  for (int i=n-1;i>0;i--) {
    swap(&arr[0], &arr[i]);
    heapify(arr,i,0);
  }
}
